import { Router } from 'express';
 
// Import Routes
import customerRoutes from '../controllers/customerController';
import productRoutes from '../controllers/productController';
import orderRoutes from '../controllers/orderController';
import loginRoutes from '../controllers/loginController';
import userRoutes from '../controllers/userController';
import { authenticateToken } from "../middlewares/authenticateToken";
import { authorizeRole } from "../middlewares/authorizeRole";
import loginController from '../controllers/loginController';
 
const router = Router();
 
// Customer's Routes
router.post("/customers", authenticateToken, customerRoutes.create);
router.get("/customers", authenticateToken, customerRoutes.getAll);
router.get("/customers/:id", authenticateToken, customerRoutes.getById);
router.put("/customers/:id", authenticateToken, customerRoutes.update);
router.delete("/customers/:id", authenticateToken, customerRoutes.delete);
 
// Product's Routes
router.post("/products", authenticateToken, productRoutes.create);
router.get("/products", authenticateToken, productRoutes.getAll);
router.get("/products/:id", authenticateToken, productRoutes.getById);
router.put("/products/:id", authenticateToken, productRoutes.update);
router.delete("/products/:id", authenticateToken, productRoutes.delete);
router.get("/productsphoto/:filename", authenticateToken, productRoutes.getUploadedFile);
 
// Order's Routes
router.post("/orders", authenticateToken, orderRoutes.createOrder);
router.get("/orders", authenticateToken, orderRoutes.getAllOrders);
router.get("/orders/:id", authenticateToken, orderRoutes.getOrderById);
router.put("/orders/:id", authenticateToken, orderRoutes.updateOrder);
 
// Login's Routes
router.post("/register", loginController.register);
router.post("/login", loginController.login);
 
router.use('/users', userRoutes);
 
export default router;
